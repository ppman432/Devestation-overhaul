			# 	higher value means faster increase

			set_variable = { global.devastation_base_gain_resistance = 0.01 }						#  must also change the loc for SDFactors1Key

			set_variable = { global.devastation_state_repair_speed_buildings_factor = -0.50 }
			
			set_variable = { global.devastation_state_division_speed = -0.35 }
			set_variable = { global.devastation_state_organization_regain = -0.20 }
			set_variable = { global.devastation_state_max_entrenchment = 0.25 }
			## set_variable = { global.devastation_state_division_attrition = 0.15 }
			
			set_variable = { global.devastation_state_local_manpower = -0.25 }

			# 	will fail to migrate and become casualties.
			set_variable = { global.casualty_multiplier_min = 0.12 }	# originally 0.45

			#   At 100% dev: exodus * factor_max (full-scale destruction = full multiplier)
			set_variable = { global.exodus_devastation_factor_min = 0.35 }
			set_variable = { global.exodus_devastation_factor_max = 1.00 }

			# During mass exodus, a much larger fraction flees in a single event.

			# Core: homeland attachment, fewer flee (Warsaw defenders, Moscow)
			set_variable = { global.exodus_core_mult_noncore_min = 0.10 }   # volume knob — lower to reduce exodus output
			set_variable = { global.exodus_core_mult_noncore_max = 0.25 }
			set_variable = { global.exodus_core_mult_iscore_min  = 0.05 }   # core territory (homeland attachment)
			set_variable = { global.exodus_core_mult_iscore_max  = 0.10 }

			#   France Exode 1940 (5% dev, summer, intact roads):       ~1.5%

			#   Stalingrad Volga evacuation (bombed, chaotic):           ~10%
			#   East Prussia Treck Jan 1945 (winter, Red Army pursuing): ~15%

			set_variable = { global.casualty_exodus_mult_t3_max = 0.20 }
			# Tier 4: dev 75-100% — total collapse, active combat during flight
			set_variable = { global.casualty_exodus_mult_t4_min = 0.14 }
			set_variable = { global.casualty_exodus_mult_t4_max = 0.28 }
			# =====================================================================
			
			 
			# Nuke casualty multipliers by state category (ALSO USED FOR DEVASTATION)
			set_variable = { global.nuke_casualty_megalopolis = 1.50 }      # +50% casualties & devastation

			set_variable = { global.nuke_casualty_large_city = 1.10 }       # +10% casualties & devastation
			set_variable = { global.nuke_casualty_city = 1.00 }             # baseline
			set_variable = { global.nuke_casualty_large_town = 0.85 }       # -15% casualties & devastation
			set_variable = { global.nuke_casualty_town = 0.70 }             # -30% casualties & devastation
			set_variable = { global.nuke_casualty_rural = 0.50 }            # -50% casualties & devastation
			set_variable = { global.nuke_casualty_pastoral = 0.35 }         # -65% casualties & devastation
			set_variable = { global.nuke_casualty_wasteland = 0.20 }        # -80% casualties & devastation
			
			# how much casualties per nuke in a state

			set_variable = { global.nuke_population_multiplier_min = 0.05 } # 5% of the population

			# Maximum population that can be killed by a single nuke
			set_variable = { global.nuke_max_population_kill_rate = 0.90 } # 90% maximum
		}
	}

			every_state = {
				add_dynamic_modifier = {

	## Used to be daily, but now weekly for performance
	on_weekly = {
		effect = {
			every_controlled_state = {
				limit = {
                    OR = {
                        # 1. Very Fast (3ms) - Immediate fail if country is at peace/stable
                        NOT = { is_controlled_by = owner } 

                        # 2. Fast (5ms) - Caught during air campaigns
                        has_recently_experienced_strat_bombing = yes 

                        # 3. Fast (6ms) - Simple variable check
                        check_variable = { state_devastation_value > 0 } 

                        # 4. Fast (9ms) - Occurs in occupied lands
                        has_resistance = yes 

                        # 5. Gatekeeper: Only run heavy combat logic if the controller is actually at war
                        # This prevents the 59ms and 116ms checks from running for neutral nations.
                        AND = {
                            controller = { has_war = yes }
                            OR = {
                                state_is_besieged = yes             # 19ms
                                state_is_combat_bubble = yes        # 33ms
                                state_isnt_fully_controlled_owner = yes # 59ms
                                neighbor_state_has_direct_war = yes # 116ms
                            }
                        }
                    }
                }
				calc_pop_change = yes
				devastation_state_calc_value = yes
				clamp_variable = { var = state_devastation_value min = 0 max = 100 }
			}
			
			calc_total_external_refugees_in_game = yes
			devastation_country_calc_value = yes
			devastation_country_calc_modifiers = yes
			devastation_country_calc_unassimilated = yes
			calc_casualties = yes
			
			if = {
				limit = {
					NOT = { has_decision = assimilation_passive_decision } 
					any_controlled_state = {
						has_unassimilated_pops = yes
					}
				}
				activate_decision = assimilation_passive_decision
			}
		}

		# =====================================================================
	}
	
	on_state_control_changed = {
		effect = {
			if = {
				limit = { has_war = yes }

				set_temp_variable = { global.state_original_controller = FROM.ID }

				FROM.FROM = {
					# =========================================================
					# Step 1: Base multiplier from flip tier.
					# state_control_changed_count read BEFORE add_to_variable,
					# so count=0 means first time captured.
					# =========================================================
					if = {
						limit = { check_variable = { state_control_changed_count < 1 } }
						set_temp_variable = { exodus_migration = global.exodus_multiplier_flip_1 }
					}
					else_if = {
						limit = { check_variable = { state_control_changed_count < 2 } }
						set_temp_variable = { exodus_migration = global.exodus_multiplier_flip_2 }
					}
					else_if = {
						limit = { check_variable = { state_control_changed_count < 3 } }
						set_temp_variable = { exodus_migration = global.exodus_multiplier_flip_3 }
					}
					else_if = {
						limit = { check_variable = { state_control_changed_count < 4 } }
						set_temp_variable = { exodus_migration = global.exodus_multiplier_flip_4 }
					}
					else_if = {
						limit = { check_variable = { state_control_changed_count < 5 } }
						set_temp_variable = { exodus_migration = global.exodus_multiplier_flip_5 }
					}
					else = {
						set_temp_variable = { exodus_migration = global.exodus_multiplier_flip_6plus }
					}

					# =========================================================
					# Step 2: Devastation intensity factor (lerp).
					# factor = factor_min + (factor_max - factor_min) * (dev/100)
					# At 0% dev -> *0.35 floor, at 100% dev -> *1.0 ceiling
					# =========================================================
					set_temp_variable = { exodus_dev_range = global.exodus_devastation_factor_max }
					subtract_from_temp_variable = { exodus_dev_range = global.exodus_devastation_factor_min }
					set_temp_variable = { exodus_dev_pct = state_devastation_value }
					divide_temp_variable = { exodus_dev_pct = 100 }
					multiply_temp_variable = { exodus_dev_range = exodus_dev_pct }
					set_temp_variable = { exodus_dev_factor = global.exodus_devastation_factor_min }
					add_to_temp_variable = { exodus_dev_factor = exodus_dev_range }
					multiply_temp_variable = { exodus_migration = exodus_dev_factor }

					# =========================================================
					# Step 3: Flag as exodus event for devastation_state_trigger_migration.
					# Replaces old sentinel check (exodus_migration == 3.0) which
					# breaks with dynamic multiplier values.
					# =========================================================
					set_temp_variable = { is_exodus_event = 1 }

					# =========================================================
					# Step 4: Fire effects. Counter increments LAST so Step 1
					# reads the pre-flip value.
					# =========================================================
					migration_get_chance = yes
					devastation_state_trigger_migration = yes
					devastation_state_trigger_factory_migration_on_capture = yes
					add_to_variable = { state_control_changed_count = 1 }
				}				
				# Factory migration must run in STATE scope (FROM), not country scope (FROM.FROM)
				# state_devastation_value, arms_factory etc. are state variables
			}
			
		}
	}
	
	on_nuke_drop = {
		effect = {
			FROM = {
				## owner = { add_war_support = -0.1 }
				
				# DIFFERENTIATE CASUALTIES BY NUKE TYPE
				if = {
					limit = { has_dynamic_modifier = { modifier = fallout_thermonuclear } }
					# Thermonuclear casualties: 25-65%
					set_temp_variable_to_random = { var = random_rate min = global.thermo_nuke_population_multiplier_min max = global.thermo_nuke_population_multiplier_max }
				}
				else = {
					# Normal nuke casualties: 5-15%
					set_temp_variable_to_random = { var = random_rate min = global.nuke_population_multiplier_min max = global.nuke_population_multiplier_max }
				}
				
				# Apply state category multiplier to casualty rate
				if = {
					limit = { is_megalopolis_state = yes }
					multiply_temp_variable = { random_rate = global.nuke_casualty_megalopolis }
				}
				else_if = {
					limit = { is_metropolis_state = yes }
					multiply_temp_variable = { random_rate = global.nuke_casualty_metropolis }
				}
				else_if = {
					limit = { is_large_city_state = yes }
					multiply_temp_variable = { random_rate = global.nuke_casualty_large_city }
				}
				else_if = {
					limit = { is_city_state = yes }
					multiply_temp_variable = { random_rate = global.nuke_casualty_city }
				}
				else_if = {
					limit = { is_large_town_state = yes }
					multiply_temp_variable = { random_rate = global.nuke_casualty_large_town }
				}
				else_if = {
					limit = { is_town_state = yes }
					multiply_temp_variable = { random_rate = global.nuke_casualty_town }
				}
				else_if = {
					limit = { is_rural_state = yes }
					multiply_temp_variable = { random_rate = global.nuke_casualty_rural }
				}
				else_if = {
					limit = { is_pastoral_state = yes }
					multiply_temp_variable = { random_rate = global.nuke_casualty_pastoral }
				}
				else_if = {
					limit = { is_wasteland_state = yes }
					multiply_temp_variable = { random_rate = global.nuke_casualty_wasteland }
				}
				
				# Cap at 90% maximum kill rate
				set_temp_variable = { max_kill_rate = global.nuke_max_population_kill_rate }
				if = {
					limit = { check_variable = { random_rate > max_kill_rate } }
					set_temp_variable = { random_rate = max_kill_rate }
				}
				
				set_temp_variable = { state_pop_deaths = state_population_k }
				multiply_temp_variable = { state_pop_deaths = random_rate }
				multiply_temp_variable = { state_pop_deaths = 1000 }
				
				if = {
					limit = { check_variable = { state_population_k > 50 } }
					set_temp_variable_to_random = { var = temp_casualty min = 30000 max = 50000 }
				}
				else_if = {
					limit = { check_variable = { state_population_k > 30 } }
					set_temp_variable_to_random = { var = temp_casualty min = 10000 max = 30000 }
				}
				else_if = {
					limit = { check_variable = { state_population_k > 10 } }
					set_temp_variable_to_random = { var = temp_casualty min = 1000 max = 10000 }
				}
				add_to_temp_variable = { temp_casualty = state_pop_deaths }
				divide_temp_variable = { temp_casualty = 1000 }
				add_to_variable = { devastation_state_casualties = temp_casualty }
				
				multiply_temp_variable = { temp_casualty = -1000 }
				add_manpower = temp_casualty
				
				set_temp_variable = { global.civ_casualty_gain = temp_casualty }
				owner = {
					set_temp_variable = { temp = global.civ_casualty_gain }
					log = "[THIS.GetTag]: Civ Casualties Gain - [?global.civ_casualty_gain]"
					add_civ_casualties_to_owner = yes
				}
				calc_pop_change = yes
		
				# DEVASTATION DIFFERENTIATION
				if = {
					limit = { has_dynamic_modifier = { modifier = fallout_thermonuclear } }
					# Thermonuclear: 45-90% devastation
					set_temp_variable_to_random = { var = nuke_devastation min = 45 max = 90 }
					set_temp_variable_to_random = { var = nuke_severity min = 3 max = 10 }
				}
				else = {
					# Normal nuke: 10-35% devastation
					set_temp_variable_to_random = { var = nuke_devastation min = 10 max = 35 }
					set_temp_variable_to_random = { var = nuke_severity min = 1 max = 3 }
				}
				
				# Apply state category multiplier to devastation (SAME AS CASUALTIES)
				if = {
					limit = { is_megalopolis_state = yes }
					multiply_temp_variable = { nuke_devastation = global.nuke_casualty_megalopolis }
				}
				else_if = {
					limit = { is_metropolis_state = yes }
					multiply_temp_variable = { nuke_devastation = global.nuke_casualty_metropolis }
				}
				else_if = {
					limit = { is_large_city_state = yes }
					multiply_temp_variable = { nuke_devastation = global.nuke_casualty_large_city }
				}
				else_if = {
					limit = { is_city_state = yes }
					multiply_temp_variable = { nuke_devastation = global.nuke_casualty_city }
				}
				else_if = {
					limit = { is_large_town_state = yes }
					multiply_temp_variable = { nuke_devastation = global.nuke_casualty_large_town }
				}
