devastation_country_calc_value = {
	# =====================================================================
	# NATIONAL DEVASTATION SCORE (NDS) — v2 FORMULA
	# =====================================================================
	# NDS = Σ(W_i × D_i) / Σ(W_i)   →  result is 0–100
	# Where:
	#   W_i = state_population_k × C_i  (population weighted by state type)
	#   D_i = state_devastation_value (0–100)
	#   C_i = state type multiplier (wasteland=5 ... megalopolis=17)
	#         reuses devastation_get_state_type_multiplier
	#
	# Because C_i appears in BOTH numerator and denominator it cannot
	# inflate the result — output is always 0–100. It only shifts the
	# weighted average toward urban states, so a bombed megalopolis
	# drags the NDS up more than an equivalent rural state.
	# =====================================================================

	clear_variable = country_devastation_value
	set_temp_variable = { nds_numerator = 0 }
	set_temp_variable = { nds_denominator = 0 }

	every_owned_state = {
		# Get C_i for this state (wasteland=5 ... megalopolis=17)
		devastation_get_state_type_multiplier = yes

		# W_i = population x C_i
		set_temp_variable = { nds_weight = state_population_k }
		multiply_temp_variable = { nds_weight = devastation_state_type_multiplier }

		# Numerator: W_i x D_i
		set_temp_variable = { nds_state_contribution = nds_weight }
		multiply_temp_variable = { nds_state_contribution = state_devastation_value }
		add_to_temp_variable = { PREV.nds_numerator = nds_state_contribution }

		# Denominator: W_i
		add_to_temp_variable = { PREV.nds_denominator = nds_weight }
	}

	# Compute NDS — guard against division by zero
	if = {
		limit = { check_variable = { nds_denominator > 0 } }
		divide_temp_variable = { nds_numerator = nds_denominator }
		set_variable = { country_devastation_value = nds_numerator }
		clamp_variable = { var = country_devastation_value min = 0 max = 100 }
	}
	else = {
		set_variable = { country_devastation_value = 0 }
	}

	##log = "[THIS.GetName]: NDS=[?country_devastation_value]"
}

	# =====================================================================
	# Each modifier = (NDS x global_scale) / 100

	# =====================================================================

	if = {

devastation_country_calc_unassimilated = {
	hidden_effect = { 
		if = {
			limit = { 
				NOT = { has_decision = assimilation_passive_decision } 
				any_controlled_state = {
					has_unassimilated_pops = yes
				}
			}
			activate_decision = assimilation_passive_decision
		}
		clear_variable = country_unassimilated_pops
		clear_variable = country_core_population_controlled
		clear_array = country_unassimilated_pops_modifier
		every_controlled_state = {
			add_to_variable = { PREV.country_unassimilated_pops = devastation_unassimilated_pops }
			
			if = {
				limit = { check_variable = { devastation_unassimilated_pops < 0 } }
				clamp_variable = { var = devastation_unassimilated_pops min = 0 }
			}
			if = {
				limit = { is_core_of = controller }
				#log = "[?THIS.GetName] is core of [?THIS.controller.GetTag]"
				add_to_temp_variable = { global.num_of_cores = 1 }
				add_to_variable = { PREV.country_core_population_controlled = state_population_k }
			}
		}
		clamp_variable = { var = country_unassimilated_pops min = 0 }
		
		
		set_variable = { country_unassimilated_pops_ratio_core = country_unassimilated_pops }
		#log = "country_unassimilated_pops_ratio_core = [?country_unassimilated_pops_ratio_core]"
		divide_variable = { country_unassimilated_pops_ratio_core = country_core_population_controlled }
		#log = "country_unassimilated_pops_ratio_core = [?country_unassimilated_pops_ratio_core]"
		multiply_variable = { country_unassimilated_pops_ratio_core = 100 }
		#log = "country_unassimilated_pops_ratio_core = [?country_unassimilated_pops_ratio_core]"
		if = {
			limit = { check_variable = { country_unassimilated_pops_ratio_core < 0.1 } check_variable = { country_unassimilated_pops > 0.1 } }
			set_variable = { country_unassimilated_pops_ratio_core = 0.099 }
		}
		
		#log = "[?THIS.GetFlag][?THIS.GetName] number of unassimilated pops is [?country_unassimilated_pops]K"
		#log = "[?THIS.GetFlag][?THIS.GetName] number of cores is [?global.num_of_cores] and have a total of [?country_core_population_controlled]K population"
		#log = "Ratio is [?country_unassimilated_pops_ratio_core|3]"
		
		
		if = {
			limit = { 
				NOT = {
					has_dynamic_modifier = {
						modifier = country_unassimilated_pops_modifier
					}
				}
			}
			add_dynamic_modifier = {
				modifier = country_unassimilated_pops_modifier
			}
		}
	
		set_temp_variable = { temp = global.unassimilated_stability_factor }
		#log = "global.unassimilated_stability_factor = [?global.unassimilated_stability_factor]"
		multiply_temp_variable = { temp = 0.1 }
		#log = "temp = [?temp]"
		multiply_temp_variable = { temp = country_unassimilated_pops_ratio_core }
		#log = "temp = [?temp]"
		add_to_array = { country_unassimilated_pops_modifier = temp }
		#log = "country_unassimilated_pops_modifier = [?country_unassimilated_pops_modifier^0|%]"
	}
}

get_devastation_gain_threshold = {
	set_temp_variable = { temp_threshold = 1 }

	if = {
		 limit = { check_variable = { base_gain > 0 } }

		if = {
			limit = { check_variable = { state_devastation_value > 94.99 } }
			subtract_from_temp_variable = { temp_threshold = global.devastation_base_gain_threshold_9 }
			#log = "Threshold 9: [?global.devastation_base_gain_threshold_9]"
		}
		else_if = {

generate_random_casualties = {
	if = {
		limit = { check_variable = { state_population_k > 1 } }
		set_temp_variable_to_random = { var = temp_casualty min = temp_min max = temp_max }
		
		set_temp_variable = { global.civ_casualty_gain = temp_casualty }
		owner = {
			set_temp_variable = { temp = global.civ_casualty_gain }
			#log = "[THIS.GetTag]: Civ Casualties Gain - [?global.civ_casualty_gain]"
			add_civ_casualties_to_owner = yes
		}
		calc_pop_change = yes
		
		multiply_temp_variable = { temp_casualty = -1 }

		
		controller = {
			set_temp_variable = { manpower = manpower_per_military_factory }
			set_temp_variable = { mils = num_of_military_factories }
		}

		add_manpower = temp_casualty
		
		controller = {
			set_temp_variable = { manpower_two = manpower_per_military_factory }
			set_temp_variable = { mils = num_of_military_factories }
			subtract_from_temp_variable = { manpower_two = manpower }
			multiply_temp_variable = { manpower_two = num_of_military_factories }
			if = {
				limit = {
					check_variable = { manpower_two > 0 }
				}
				multiply_temp_variable = { manpower_two = -1 }
				add_manpower = manpower_two
			}
		}

		multiply_temp_variable = { temp_casualty = -1 }
		set_temp_variable = { devastation_state_casualties_k = temp_casualty }
		divide_temp_variable = { devastation_state_casualties_k = 1000 }
		add_to_variable = { devastation_state_casualties = devastation_state_casualties_k }
	}
}

calc_casualties = {
	# =====================================================================
	# Military column uses persistent_mil_casualties_k (never resets).
	# Accumulated in casualties_on_actions.txt at 30% of weekly delta.
	# casualties_k (native HOI4) resets to 0 on peace — not used here.
	# =====================================================================
	if = {
		limit = { check_variable = { persistent_mil_casualties_k > 999.999 } }
		set_variable = { persistent_mil_casualties_m = persistent_mil_casualties_k }
		divide_variable = { persistent_mil_casualties_m = 1000 }
	}
	else = {
		clear_variable = persistent_mil_casualties_m
	}
	if = {
		limit = { check_variable = { civ_casualties_k > 999.999 } }
		set_variable = { civ_casualties_m = civ_casualties_k }
		divide_variable = { civ_casualties_m = 1000 }
	}
	else = {
		clear_variable = civ_casualties_m
	}
	
	# Total = persistent military deaths + civilian deaths
	set_variable = { total_casualties_k = persistent_mil_casualties_k }
	add_to_variable = { total_casualties_k = civ_casualties_k }
	if = {
		limit = { check_variable = { total_casualties_k > 999.999 } }
		set_variable = { total_casualties_m = total_casualties_k }
		divide_variable = { total_casualties_m = 1000 }
	}
	else = {
		clear_variable = total_casualties_m
	}
}

devastation_state_trigger_migration = {
	if = {
		limit = {
			# if population is greater than 1k, then migrate
			check_variable = { state_population_k > 1 }
			#has_state_devastation = yes
		}
		set_temp_variable = { devastation_state_refugee = 0 }

		# Exodus vs normal migration branch.
		# is_exodus_event = 1 set by on_state_control_changed only.
		# For exodus: dev% + 25 floor so even 0%-dev captures cause some flight.
		# For normal migration: raw devastation, exodus_migration defaults to 1.
		if = {
			limit = { check_variable = { is_exodus_event = 1 } }
			set_temp_variable = { devastation_percentage = state_devastation_value }
			add_to_temp_variable = { devastation_percentage = 25 }
			clamp_temp_variable = { var = devastation_percentage min = 25 max = 100 }
		}
		else = {
			set_temp_variable = { devastation_percentage = state_devastation_value }
			set_temp_variable = { exodus_migration = 1 }
		}
		divide_temp_variable = { devastation_percentage = 100 }
		#log = "[THIS.GetName]: Devastation Percentage - [?devastation_percentage|%]"
		
		devastation_get_migration_type_multiplier = yes
		#log = "[THIS.GetName]: Migration Category Multiplier - [?devastation_migration_type_multiplier]"
		
		set_temp_variable = { devastation_state_refugee = state_population_k }
		#log = "[THIS.GetName]: State Population - [?state_population_k] K"
		
		multiply_temp_variable = { devastation_state_refugee = devastation_percentage }
		multiply_temp_variable = { devastation_state_refugee = devastation_migration_type_multiplier }
		
		# Core multiplier: exodus uses much higher ranges than weekly passive migration.
		# Weekly: 0.01-0.05 (core) / 0.01-0.10 (non-core) — slow background drain.
		# Exodus: 0.15-0.35 (core) / 0.40-0.80 (non-core) — mass panic event.
		if = {
			limit = { check_variable = { is_exodus_event = 1 } }
			if = {
				limit = { is_core_of = controller }
				set_temp_variable_to_random = { var = devastation_state_core_multiplier min = global.exodus_core_mult_iscore_min max = global.exodus_core_mult_iscore_max }
			}
			else = {
				set_temp_variable_to_random = { var = devastation_state_core_multiplier min = global.exodus_core_mult_noncore_min max = global.exodus_core_mult_noncore_max }
			}
		}
		else = {
			if = {
				limit = { is_core_of = controller }
				set_temp_variable_to_random = { var = devastation_state_core_multiplier min = 0.01 max = 0.05 }
			}
			else = {
				set_temp_variable_to_random = { var = devastation_state_core_multiplier min = 0.01 max = 0.1 }
			}
		}
		multiply_temp_variable = { devastation_state_refugee = devastation_state_core_multiplier }
		#log = "[THIS.GetName]: Core Multiplier - [?devastation_state_core_multiplier]"
	
		if = {
			limit = { check_variable = { exodus_migration = 0 } }
			set_temp_variable = { exodus_migration = 1 }
		}
		multiply_temp_variable = { devastation_state_refugee = exodus_migration }
		
		if = {
			limit = { check_variable = { devastation_state_refugee > 2147.483 } }
			
			# if # of refugee is greater than 2 million, clamp to 2 million
			
			#log = "[THIS.GetName]: Refugee wil overflow! Clamping to 2 million"
			set_temp_variable = { devastation_state_refugee = 2147.483 }
			multiply_temp_variable = { devastation_state_refugee = 1000 }
			multiply_temp_variable = { devastation_state_refugee = -1 }
			#log = "[THIS.GetName]: Refugees Lost - [?devastation_state_refugee]"

			#effect_tooltip = {
				add_manpower = devastation_state_refugee
			#}
			
			# if # of refugee is greater than 2 million, do a for-loop
			
			##log = "[THIS.GetName]: Refugee wil overflow! Proceeding to For-Loop 10x"
			for_loop_effect = {
				start = 0
				end = 10
				value = loop_value
				#log = "[THIS.GetName]: Loop Value is [?loop_value]"
				##log = "[THIS.GetName]: Refugees - [?devastation_state_refugee]"
				
				multiply_temp_variable = { devastation_state_refugee = -1 }
			#	effect_tooltip = {
					add_manpower = devastation_state_refugee
			#	}
			}
		}
		else = {
			multiply_temp_variable = { devastation_state_refugee = 1000 }
			#log = "[THIS.GetName]: Refugees Lost - [?devastation_state_refugee]"
			
			multiply_temp_variable = { devastation_state_refugee = -1 }

			#effect_tooltip = {
				add_manpower = devastation_state_refugee
			#}
		}
		
		multiply_temp_variable = { devastation_state_refugee = -1 }
		set_temp_variable = { casualties = devastation_state_refugee }
		
		# Casualty rate: exodus scales with devastation, weekly migration uses flat rate.
		if = {
			limit = { check_variable = { is_exodus_event = 1 } }
			if = {
				limit = { check_variable = { state_devastation_value < 20 } }
				# Tier 1 (0-20%): swift capture, intact roads (e.g. France 1940: ~1.5%)
				set_temp_variable_to_random = { var = temp_casualty_multiplier min = global.casualty_exodus_mult_t1_min max = global.casualty_exodus_mult_t1_max }
			}
			else_if = {
				limit = { check_variable = { state_devastation_value < 50 } }
				# Tier 2 (20-50%): contested, partial collapse (e.g. Kharkov: ~6.7%)
				set_temp_variable_to_random = { var = temp_casualty_multiplier min = global.casualty_exodus_mult_t2_min max = global.casualty_exodus_mult_t2_max }
			}
			else_if = {
				limit = { check_variable = { state_devastation_value < 75 } }
				# Tier 3 (50-75%): heavy bombing, roads destroyed (e.g. Stalingrad: ~10%)
				set_temp_variable_to_random = { var = temp_casualty_multiplier min = global.casualty_exodus_mult_t3_min max = global.casualty_exodus_mult_t3_max }
			}
			else = {
				# Tier 4 (75-100%): total collapse, active combat during flight (~15-25%)
				set_temp_variable_to_random = { var = temp_casualty_multiplier min = global.casualty_exodus_mult_t4_min max = global.casualty_exodus_mult_t4_max }
			}
		}
		else = {
			set_temp_variable_to_random = { var = temp_casualty_multiplier min = global.casualty_multiplier_min max = global.casualty_multiplier_max }
		}
		multiply_temp_variable = { casualties = temp_casualty_multiplier }
		#log = "[THIS.GetName]: Refugees Becoming Casulaties - [?temp_casualty_multiplier|%] = [?casualties]"
		
		subtract_from_temp_variable = { devastation_state_refugee = casualties }
		set_temp_variable = { devastation_state_casualties_k = casualties }
		divide_temp_variable = { devastation_state_casualties_k = 1000 }
		add_to_variable = { devastation_state_casualties = devastation_state_casualties_k }
		
		round_temp_variable = devastation_state_casualties
		round_temp_variable = devastation_state_refugee
		
		set_temp_variable = { devastation_state_refugee_k = devastation_state_refugee }
		divide_temp_variable = { devastation_state_refugee_k = 1000 }
		add_to_variable = { devastation_state_refugee_lost = devastation_state_refugee_k }
				
		set_temp_variable = { global.civ_casualty_gain = casualties }
		owner = {
			set_temp_variable = { temp = global.civ_casualty_gain }
			#log = "[THIS.GetTag]: Civ Casualties Gain - [?global.civ_casualty_gain]"
			add_civ_casualties_to_owner = yes
		}
		
		# divide the refugees in groups of 10%, 20%, 30%, and 40%
		set_temp_variable = { refugee_gain_1 = devastation_state_refugee }
		multiply_temp_variable = { refugee_gain_1 = 0.1 }
		set_temp_variable = { refugee_gain_2 = devastation_state_refugee }
		multiply_temp_variable = { refugee_gain_2 = 0.2 }
		set_temp_variable = { refugee_gain_3 = devastation_state_refugee }
		multiply_temp_variable = { refugee_gain_3 = 0.3 }
		set_temp_variable = { refugee_gain_4 = devastation_state_refugee }
		multiply_temp_variable = { refugee_gain_4 = 0.4 }
		round_temp_variable = refugee_gain_1
		round_temp_variable = refugee_gain_2
		round_temp_variable = refugee_gain_3
		round_temp_variable = refugee_gain_4
		
		#set_temp_variable = { global.state_original_controller = FRA }
		calc_pop_change = yes

		# =====================================================================
		# REFUGEE DESTINATION ROUTING — PERMANENT RESETTLEMENT MODEL

# =====================================================================
# devastation_state_trigger_factory_migration_on_capture
#
# Fires on on_state_control_changed, in STATE scope (FROM = captured state).
# Reads state_devastation_value and state_control_changed_count from state scope.
# Stores global.fac_mig_origin (state ref) for var: access in nested scopes.
# Stores global.fac_mig_owner (numeric country ID) for war-check filters.
#
# STEP 1 — Gate: devastation > 10 and factories exist.
# STEP 2 — Compute at-risk count using 9-tier ratio based on devastation breakpoints.
#           Flip count applies a small penalty (max -20%), halved if dev >= 60.
#           Floor: at least 1 factory at risk if dev > 30 and factories exist.
# STEP 3 — Loop once per at-risk factory. Each iteration rolls independently:
#           Outcome weights shift across 9 devastation tiers:
#             A) Migrate to neighbor (friendly-controlled neighboring state)
#             B) Flee far (random state within 1500km, not at war with owner;
#                includes owner's own distant states (>400km) and foreign states)
#             C) Destroyed in place
#             D) Nothing happens
#           Each outcome picks mil vs civ 50/50 (factor=0 guard if none available).
# STEP 4 — calc_total_factory_migration called on origin state to update tracking vars.
# =====================================================================
devastation_state_trigger_factory_migration_on_capture = {

	# -----------------------------------------------------------------
	# STEP 1 — Gate checks
	# Scope: STATE (the captured state = THIS throughout)
	# Inside random_neighbor_state / random_state: PREV = THIS = origin state
	# Inside random_neighbor_state / random_state: PREV = state, PREV.PREV = old controller country
	# In outer random_list modifier context: PREV = old controller country (FROM)
	# -----------------------------------------------------------------
	if = {
		limit = {
			check_variable = { state_devastation_value > 10 }
			OR = {
				arms_factory > 0
				industrial_complex > 0
			}
		}

		# -----------------------------------------------------------------
		# STEP 2 — Compute at-risk factory count
		# -----------------------------------------------------------------
		set_temp_variable = { fac_total = arms_factory }
		add_to_temp_variable = { fac_total = industrial_complex }

		# 9-tier at-risk ratio
		if =      { limit = { check_variable = { state_devastation_value < 10 } } set_temp_variable = { fac_at_risk_ratio = 0.15 } }
		else_if = { limit = { check_variable = { state_devastation_value < 20 } } set_temp_variable = { fac_at_risk_ratio = 0.28 } }
		else_if = { limit = { check_variable = { state_devastation_value < 30 } } set_temp_variable = { fac_at_risk_ratio = 0.45 } }
		else_if = { limit = { check_variable = { state_devastation_value < 45 } } set_temp_variable = { fac_at_risk_ratio = 0.60 } }
		else_if = { limit = { check_variable = { state_devastation_value < 60 } } set_temp_variable = { fac_at_risk_ratio = 0.72 } }
		else_if = { limit = { check_variable = { state_devastation_value < 75 } } set_temp_variable = { fac_at_risk_ratio = 0.83 } }
		else_if = { limit = { check_variable = { state_devastation_value < 85 } } set_temp_variable = { fac_at_risk_ratio = 0.90 } }
		else_if = { limit = { check_variable = { state_devastation_value < 95 } } set_temp_variable = { fac_at_risk_ratio = 0.95 } }
		else =    { set_temp_variable = { fac_at_risk_ratio = 1.0 } }

		# No flip count penalty — devastation level is the sole driver of at-risk ratio.

		set_temp_variable = { fac_at_risk = fac_total }
		multiply_temp_variable = { fac_at_risk = fac_at_risk_ratio }
		round_temp_variable = fac_at_risk

		if = {
			limit = {
				check_variable = { fac_at_risk < 1 }
				check_variable = { state_devastation_value > 30 }
			}
			set_temp_variable = { fac_at_risk = 1 }
		}

		#log = "[THIS.GetName]: [?fac_total] facs | [?state_devastation_value]% dev | [?fac_at_risk] at risk"

		# -----------------------------------------------------------------
		# STEP 3 — Per-factory loop
		# Scope = STATE (origin captured state) throughout.
		# Inside random_neighbor_state / random_state: PREV = origin state.
		# -----------------------------------------------------------------
		if = {
			limit = { check_variable = { fac_at_risk > 0 } }

			for_loop_effect = {
				start = 0
				end = fac_at_risk
				value = fac_loop_index

				# Weights per tier (neighbor | flee far | destroyed | nothing):
				# T1 <10%:   15 | 3  | 5  | 77
				# T2 10-20%: 25 | 5  | 10 | 60
				# T3 20-30%: 35 | 8  | 18 | 39
				# T4 30-45%: 38 | 10 | 27 | 25
				# T5 45-60%: 30 | 12 | 38 | 20
				# T6 60-75%: 22 | 10 | 50 | 18
				# T7 75-85%: 14 | 8  | 62 | 16
				# T8 85-95%: 8  | 5  | 72 | 15
				# T9 >95%:   5  | 3  | 80 | 12
				random_list = {

					# -------------------------------------------------------
					# OUTCOME A — Migrate to a neighboring state
					# PREV = origin state. Add factory to THIS (neighbor), remove from PREV.
					# -------------------------------------------------------
					15 = {
						modifier = { add = 10 check_variable = { state_devastation_value > 9  } }
						modifier = { add = 10 check_variable = { state_devastation_value > 19 } }
						modifier = { add = 3  check_variable = { state_devastation_value > 29 } }
						modifier = { add = -8 check_variable = { state_devastation_value > 44 } }
						modifier = { add = -8 check_variable = { state_devastation_value > 59 } }
						modifier = { add = -8 check_variable = { state_devastation_value > 74 } }
						modifier = { add = -6 check_variable = { state_devastation_value > 84 } }
						modifier = { add = -3 check_variable = { state_devastation_value > 94 } }
						modifier = {
							factor = 0
							NOT = { any_neighbor_state = { owner = { NOT = { has_war_with = PREV } } } }
						}
						random_neighbor_state = {
							limit = { owner = { NOT = { has_war_with = PREV.PREV } } }
							random_list = {
								50 = {
									modifier = { factor = 0 PREV = { arms_factory < 1 } }
									add_building_construction = { type = arms_factory level = 1 instant_build = yes }
									add_to_variable = { devastation_state_mil_gain = 1 }
									PREV = {
										remove_building = { type = arms_factory level = 1 }
										add_to_variable = { devastation_state_mil_lost = 1 }
									}
								}
								50 = {
									modifier = { factor = 0 PREV = { industrial_complex < 1 } }
									add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
									add_to_variable = { devastation_state_civ_gain = 1 }
									PREV = {
										remove_building = { type = industrial_complex level = 1 }
										add_to_variable = { devastation_state_civ_lost = 1 }
									}
								}
							}
						}
					}

					# -------------------------------------------------------
					# OUTCOME B — Flee far (random state within 1500km)
					# PREV = origin state for distance_to target and building ops.
					# -------------------------------------------------------
					3 = {
						modifier = { add = 2  check_variable = { state_devastation_value > 9  } }
						modifier = { add = 3  check_variable = { state_devastation_value > 19 } }
						modifier = { add = 2  check_variable = { state_devastation_value > 29 } }
						modifier = { add = 2  check_variable = { state_devastation_value > 44 } }
						modifier = { add = -2 check_variable = { state_devastation_value > 59 } }
						modifier = { add = -2 check_variable = { state_devastation_value > 74 } }
						modifier = { add = -3 check_variable = { state_devastation_value > 84 } }
						modifier = { add = -2 check_variable = { state_devastation_value > 94 } }
						modifier = {
							factor = 0
							NOT = {
								any_state = {
									distance_to = { value < 1500 target = THIS }
									owner = { NOT = { has_war_with = PREV } }
									OR = {
										NOT = { is_owned_by = PREV }
										AND = {
											is_owned_by = PREV
											distance_to = { value > 400 target = THIS }
										}
									}
								}
							}
						}
						random_state = {
							limit = {
								distance_to = { value < 1500 target = PREV }
								owner = { NOT = { has_war_with = PREV.PREV } }
								OR = {
									NOT = { is_owned_by = PREV.PREV }
									AND = {
										is_owned_by = PREV.PREV
										distance_to = { value > 400 target = PREV }
									}
								}
							}
							random_list = {
								50 = {
									modifier = { factor = 0 PREV = { arms_factory < 1 } }
									add_building_construction = { type = arms_factory level = 1 instant_build = yes }
									add_to_variable = { devastation_state_mil_gain = 1 }
									PREV = {
										remove_building = { type = arms_factory level = 1 }
										add_to_variable = { devastation_state_mil_lost = 1 }
									}
								}
								50 = {
									modifier = { factor = 0 PREV = { industrial_complex < 1 } }
									add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
									add_to_variable = { devastation_state_civ_gain = 1 }
									PREV = {
										remove_building = { type = industrial_complex level = 1 }
										add_to_variable = { devastation_state_civ_lost = 1 }
									}
								}
							}
						}
					}

					# -------------------------------------------------------
					# OUTCOME C — Destroyed in place
					# Scope is THIS (origin state) — no PREV needed.
					# -------------------------------------------------------
					5 = {
						modifier = { add = 5  check_variable = { state_devastation_value > 9  } }
						modifier = { add = 8  check_variable = { state_devastation_value > 19 } }
						modifier = { add = 9  check_variable = { state_devastation_value > 29 } }
						modifier = { add = 11 check_variable = { state_devastation_value > 44 } }
						modifier = { add = 12 check_variable = { state_devastation_value > 59 } }
						modifier = { add = 12 check_variable = { state_devastation_value > 74 } }
						modifier = { add = 10 check_variable = { state_devastation_value > 84 } }
						modifier = { add = 8  check_variable = { state_devastation_value > 94 } }
						random_list = {
							50 = {
								modifier = { factor = 0 arms_factory < 1 }
								remove_building = { type = arms_factory level = 1 }
								add_to_variable = { devastation_state_mil_lost = 1 }
							}
							50 = {
								modifier = { factor = 0 industrial_complex < 1 }
								remove_building = { type = industrial_complex level = 1 }
								add_to_variable = { devastation_state_civ_lost = 1 }
							}
						}
					}

					# -------------------------------------------------------
					# OUTCOME D — Nothing happens
					# Base 40 (T1). Aggressively cut with devastation.
					# T1<10%: 40 | T2: 28 | T3: 18 | T4: 12 | T5: 8 | T6: 6 | T7: 5 | T8: 4 | T9: 3
					# -------------------------------------------------------
					40 = {
						modifier = { add = -12 check_variable = { state_devastation_value > 9  } }
						modifier = { add = -10 check_variable = { state_devastation_value > 19 } }
						modifier = { add = -6  check_variable = { state_devastation_value > 29 } }
						modifier = { add = -4  check_variable = { state_devastation_value > 44 } }
						modifier = { add = -2  check_variable = { state_devastation_value > 59 } }
						modifier = { add = -1  check_variable = { state_devastation_value > 74 } }
						modifier = { add = -1  check_variable = { state_devastation_value > 84 } }
						modifier = { add = -1  check_variable = { state_devastation_value > 94 } }
					}
				}
			}
		}

		# -----------------------------------------------------------------
		# STEP 4 — Update tracking variables on origin state
		# -----------------------------------------------------------------
		calc_total_factory_migration = yes
	}
}

calc_total_factory_migration = {
	set_variable = { devastation_state_factory_gain = devastation_state_civ_gain }
	add_to_variable = { devastation_state_factory_gain = devastation_state_mil_gain }
	
	set_variable = { devastation_state_factory_lost = devastation_state_civ_lost }
	add_to_variable = { devastation_state_factory_lost = devastation_state_mil_lost }
	
	set_variable = { devastation_state_factory_effective_gain = devastation_state_factory_gain }
	subtract_from_variable = { devastation_state_factory_effective_gain = devastation_state_factory_lost }
	
	set_variable = { devastation_state_civ_effective_gain = devastation_state_civ_gain }
	subtract_from_variable = { devastation_state_civ_effective_gain = devastation_state_civ_lost }
	
	set_variable = { devastation_state_mil_effective_gain = devastation_state_mil_gain }
	subtract_from_variable = { devastation_state_mil_effective_gain = devastation_state_mil_lost }
}

devastation_state_calc_value = {
	calc_pop_change = yes
	set_variable = { state_devastation_previous_value = state_devastation_value }
	#log = "[THIS.GetName] State Devastation Previous Value: [?state_devastation_previous_value]"
	
	set_temp_variable = { base_gain = 0 }
	
	set_variable = { max_resistance_state_cap = resistance }
	multiply_variable = { max_resistance_state_cap = 0.5 }
	
	#log = "[THIS.GetName] Base Gain: [?base_gain]"
	
	#if = {

			#log = "[THIS.GetName] Base Gain Resistance: [?base_gain_resistance]"
			
			#divide_temp_variable = { base_gain_resistance = 333.333 }
			multiply_temp_variable = { base_gain_resistance = global.devastation_base_gain_resistance }
			divide_temp_variable = { base_gain_resistance = 1 }			### Added this, otherwise resistance is too strong and cannot be fought devastation-wise
			set_variable = { state_devastation_base_gain_resistance = base_gain_resistance }
			#log = "[THIS.GetName] Base Gain Resistance after multiplier: [?base_gain_resistance]"
			
			add_to_temp_variable = { base_gain = base_gain_resistance }
			#log = "[THIS.GetName] Base Gain Resistance -> Base Gain: [?base_gain]"
		}
	#}
	#else = {
		if = {
			limit = {
				controller_has_war_with_owner = yes
				state_is_fully_controlled_owner = no
			}
			
			if = {
				limit = { check_variable = { state_devastation_value < 56 } }
				add_to_temp_variable = { base_gain = global.devastation_base_gain_controller_at_war_with_owner }
				#log = "[THIS.GetName] Global|Controller at war with owner: [?global.devastation_base_gain_controller_at_war_with_owner] -> Base Gain: [?base_gain]"
			}
			else = {
				add_to_temp_variable = { base_gain = 0.05 }
				#log = "[THIS.GetName] Base Gain: [?base_gain]"
			}
		}
		
		# Partial occupation
		if = {
			limit = {
				state_is_partially_occupied = yes
				state_isnt_fully_controlled_owner = yes
				check_variable = { state_devastation_value < 91 }
			}
			add_to_temp_variable = { base_gain = global.devastation_base_gain_partial_occupation }
		}

		# Combat bubble
		if = {
			limit = { 
				state_is_combat_bubble = yes 
				state_isnt_fully_controlled_owner = yes
				check_variable = { state_devastation_value < 51 }
			}
			add_to_temp_variable = { base_gain = global.devastation_base_gain_partial_occupation }
		}
		
		if = {
			limit = {
				neighbor_state_has_direct_war = yes
				state_isnt_fully_controlled_owner = no
				check_variable = { state_devastation_value < 36 }
			}
			add_to_temp_variable = { base_gain = global.devastation_base_gain_neighbor_state_is_enemy }
			#log = "[THIS.GetName] Global|Neighbor state is enemy: [?global.devastation_base_gain_neighbor_state_is_enemy] -> Base Gain: [?base_gain]"
		}
	#}
	
	if = {
		limit = {
			has_recently_experienced_strat_bombing = yes
		}
		add_to_temp_variable = { base_gain = global.devastation_base_gain_experienced_bombing }
		#log = "[THIS.GetName] Global|Bombing: [?global.devastation_base_gain_controller_at_war_with_owner] -> Base Gain: [?base_gain]"
		
		#set_temp_variable = { temp_min = 250 }		# Original 50
		#set_temp_variable = { temp_max = 1000 }		# Original 201
		#generate_random_casualties = yes
		#devastation_state_trigger_migration = yes
		migration_get_chance = yes
	}
	
	# SIEGE DEVASTATION
	if = {
		limit = { state_is_besieged = yes }
    
		add_to_temp_variable = { base_gain = global.devastation_base_gain_siege }  # +2.5% devastation per week
	}
	
	set_temp_variable = { base_loss = 0 }
	#log = "[THIS.GetName] Base Loss: [?base_loss]"
	
	if = {
		limit = {
			controller_has_war_with_owner = no
			state_isnt_fully_controlled_owner = no
			neighbor_state_has_direct_war = no
		}
		add_to_temp_variable = { base_loss = global.devastation_base_gain_controller_at_peace_with_owner }
		#log = "[THIS.GetName] Global|Controller at peace with owner: [?global.devastation_base_gain_controller_at_war_with_owner] -> Base Loss: [?base_loss]"
	}
	if = {
		limit = {
			controller_at_peace = yes
		}
		add_to_temp_variable = { base_loss = global.devastation_base_gain_controller_at_peace }
		#log = "[THIS.GetName] Global|Controller at peace: [?global.devastation_base_gain_controller_at_war_with_owner] -> Base Loss: [?base_loss]"
	}
	
	controller = {
		if = {
			limit = { NOT = { check_variable = { modifier@state_devastation_reduction = 0 } } }
			
			PREV = {
				add_to_temp_variable = { base_loss = PREV.modifier@state_devastation_reduction }
				#log = "[THIS.GetName] Modifier|State Devastation Reduction: [?PREV.modifier@state_devastation_reduction] -> Base Loss: [?base_loss]"
			}
		}
		if = {
			limit = { NOT = { check_variable = { modifier@state_devastation_reduction_factor = 0 } } }
			
			PREV = {
				set_temp_variable = { temp_a = base_loss }
				set_temp_variable = { temp_b = PREV.modifier@state_devastation_reduction_factor }
				multiply_temp_variable = { temp_b = -1 }
				multiply_temp_variable = { temp_a = temp_b }
				add_to_temp_variable = { base_loss = temp_a }
				#log = "[THIS.GetName] Modifier|State Devastation Reduction Factor: [?PREV.modifier@state_devastation_reduction] -> Base Loss: [?base_loss]"
			}
		}
	}
	
	# Apply state category gain multiplier
	devastation_get_state_category_gain_multiplier = yes
	multiply_temp_variable = { base_gain = state_category_gain_mult }
	multiply_temp_variable = { base_loss = state_category_gain_mult }
	#log = "[THIS.GetName] Base Gain After State Category Multiplier: [?base_gain]"
	
	
	#log = "[THIS.GetName] Base Gain Before Threshold: [?base_gain]"
	get_devastation_gain_threshold = yes
	multiply_temp_variable = { base_gain = temp_threshold }
	#log = "[THIS.GetName] Base Gain After Threshold: [?base_gain]"
	set_temp_variable = { temp_threshold_loss = 1 }					# Addition, saturation factor helps reducing devastation now
	subtract_from_temp_variable = { temp_threshold_loss = temp_threshold }					# Addition
	add_to_temp_variable = { temp_threshold_loss = 1 }					# Addition
	multiply_temp_variable = { base_loss = temp_threshold_loss }			# Addition
	
	# =====================================================================
	# WARFARE CONDUCT — DEVASTATION GAIN MULTIPLIER
	# Only fires when base_gain > 0 (devastation actively increasing).
	# Does NOT touch base_loss — only scales the gain side.
	# Uses controller scope to copy dev_mult into a temp var first,
	# since multiply_temp_variable cannot read cross-scope variables directly.
	# =====================================================================
	if = {
		limit = {
			check_variable = { base_gain > 0 }
			controller = { check_variable = { warfare_stance_dev_mult > 0.1 } }
		}
		# Copy country variable into state-scope temp var before multiplying
		controller = { set_temp_variable = { global.bm_dev_mult_temp = warfare_stance_dev_mult } }
		multiply_temp_variable = { base_gain = global.bm_dev_mult_temp }
	}
	
	add_to_variable = { state_devastation_value = base_gain }
	add_to_variable = { state_devastation_value = base_loss }
	
	set_variable = { state_devastation_value_change = base_gain }
	add_to_variable = { state_devastation_value_change = base_loss }
	
	devastation_state_calc_modifiers = yes
	
	if = {
		limit = { state_isnt_fully_controlled_owner = yes }
		
		set_temp_variable = { global.state_original_controller = THIS.owner }
		random = {
			chance = state_devastation_value
			devastation_state_trigger_migration = yes
			#random_country = { limit = { is_ai = no } country_event = devastation_event.0 }
		}
		set_temp_variable = { temp_min = 50 }
		set_temp_variable = { temp_max = 100 }
		generate_random_casualties = yes
	}
	
	if = {
		limit = {
			state_is_fully_controlled_owner = yes
			check_variable = { resistance > 0 }
		}

		# -- base gain from resistance (for devastation calculation) --
		set_temp_variable = { base_gain_resistance = resistance }
		round_temp_variable = base_gain_resistance
		multiply_temp_variable = { base_gain_resistance = global.devastation_base_gain_resistance }
		divide_temp_variable = { base_gain_resistance = 1 }    # keeps resistance balanced
		set_variable = { state_devastation_base_gain_resistance = base_gain_resistance }
		add_to_temp_variable = { base_gain = base_gain_resistance }
		
		# ===================================================================
		# NEW OCCUPATION CASUALTY SYSTEM
		# ===================================================================
		
		# STEP 1: Calculate base weekly death rate from population
		# Formula: population_k * base_rate
		# This gives us a baseline: larger populations = more deaths possible
		set_temp_variable = { occupation_death_base = state_population_k }
		multiply_temp_variable = { occupation_death_base = 0.01 }  # Base weekly death rate (0.01% of population)
		
		# STEP 2: Resistance multiplier (higher resistance = more reprisals and deaths)
		# resistance ranges 0-100, we scale it to 0.1x - 6.0x multiplier
		# At 0% resistance = 0.1x (minimal deaths - natural/accidental)
		# At 100% resistance = 6.0x (maximum reprisals and brutality)
		set_temp_variable = { resistance_factor = resistance }
		divide_temp_variable = { resistance_factor = 100 }  # Normalize to 0.0-1.0
		multiply_temp_variable = { resistance_factor = 5.9 }   # Scale to 0.0-5.9
		add_to_temp_variable = { resistance_factor = 0.1 }     # Make it 0.1-6.0x multiplier
		
		multiply_temp_variable = { occupation_death_base = resistance_factor }
		
		# STEP 3: Devastation multiplier (destroyed areas = disease, starvation, chaos)
		# devastation ranges 0-100, we scale it to 1.0x - 4.0x multiplier
		set_temp_variable = { devastation_factor = state_devastation_value }
		divide_temp_variable = { devastation_factor = 100 }  # Normalize to 0.0-1.0
		multiply_temp_variable = { devastation_factor = 3 }   # Scale to 0.0-3.0
		add_to_temp_variable = { devastation_factor = 1 }     # Make it 1.0-4.0x multiplier
		
		multiply_temp_variable = { occupation_death_base = devastation_factor }
		
		# STEP 4: State category factor (urban = easier to control/kill, rural = harder to find people)
		set_temp_variable = { state_category_factor = 1.0 }
		
		if = { limit = { is_megalopolis_state = yes } set_temp_variable = { state_category_factor = 1.8 } }  # Dense population, easy to round up
		else_if = { limit = { is_metropolis_state = yes } set_temp_variable = { state_category_factor = 1.5 } }
		else_if = { limit = { is_large_city_state = yes } set_temp_variable = { state_category_factor = 1.3 } }
		else_if = { limit = { is_city_state = yes } set_temp_variable = { state_category_factor = 1.0 } }  # Baseline
		else_if = { limit = { is_large_town_state = yes } set_temp_variable = { state_category_factor = 0.85 } }
		else_if = { limit = { is_town_state = yes } set_temp_variable = { state_category_factor = 0.70 } }
		else_if = { limit = { is_rural_state = yes } set_temp_variable = { state_category_factor = 0.50 } }  # Harder to find people
		else_if = { limit = { is_pastoral_state = yes } set_temp_variable = { state_category_factor = 0.30 } }  # Very spread out
		else_if = { limit = { is_wasteland_state = yes } set_temp_variable = { state_category_factor = 0.15 } }  # Almost nobody there
		
		multiply_temp_variable = { occupation_death_base = state_category_factor }
		
		# STEP 5: Occupation law multiplier
		set_temp_variable = { occupation_casualty_factor = 1.0 }

		### VERY HARSH / OPPRESSIVE (high civilian death, deportations, executions)
		if = { limit = { occupation_law = brutally_oppressive_occupation } set_temp_variable = { occupation_casualty_factor = 6.0 } }  # Nazi Poland / Belarus / Yugoslavia-level brutality
		if = { limit = { occupation_law = forced_labor_occupation } set_temp_variable = { occupation_casualty_factor = 4.5 } }       # Widespread deportations, starvation, but not extermination policy
		if = { limit = { occupation_law = harsh_quotas_occupation } set_temp_variable = { occupation_casualty_factor = 3.0 } }        # Harsh exploitation (Ukraine 1942, China 1939)
		if = { limit = { occupation_law = martial_law_occupation } set_temp_variable = { occupation_casualty_factor = 1.75 } }        # Curfews, shootings of partisans; Italy 1944-level

		### MILITARY / STRONG CONTROL (moderate casualties, pragmatic governance)
		if = { limit = { occupation_law = military_governor_occupation } set_temp_variable = { occupation_casualty_factor = 1.0 } }   # Baseline case
		if = { limit = { occupation_law = secret_police_oversight } set_temp_variable = { occupation_casualty_factor = 0.85 } }        # Gestapo/NKVD-style repression

		### LENIENT / SEMI-AUTONOMOUS (low casualties, collaborationist governments)
		if = { limit = { occupation_law = autonomous_occupation } set_temp_variable = { occupation_casualty_factor = 0.6 } }          # Vichy-like arrangement
		if = { limit = { occupation_law = foreign_civilian_oversight } set_temp_variable = { occupation_casualty_factor = 0.45 } }    # Allied military govt in Germany 1945–49
		if = { limit = { occupation_law = reconciliation } set_temp_variable = { occupation_casualty_factor = 0.35 } }                 # Japan's occupation of Taiwan 1920s-30s
		if = { limit = { occupation_law = independent_rule } set_temp_variable = { occupation_casualty_factor = 0.3 } }                # Puppet regimes (Manchukuo, Burma)
		if = { limit = { occupation_law = local_police_force_garrison } set_temp_variable = { occupation_casualty_factor = 0.8 } }    # Indigenous repression—low stability, still bloody
		if = { limit = { occupation_law = liberate_workers_occupation } set_temp_variable = { occupation_casualty_factor = 0.7 } }    # "Red liberation" with purges, limited but real death tolls
		if = { limit = { occupation_law = integration_occupation } set_temp_variable = { occupation_casualty_factor = 0.5 } }         # Post-annexation assimilation (Alsace-Lorraine, 1941 Croatia)

		### COLONIAL / EXPLOITATIVE (chronic but lower-intensity deaths)
		if = { limit = { occupation_law = colonial_police } set_temp_variable = { occupation_casualty_factor = 1.5 } }                 # Everyday brutality (Congo Free State–style)
		if = { limit = { occupation_law = colonial_police_improved } set_temp_variable = { occupation_casualty_factor = 1.2 } }        # Late colonial reforms, still repressive
		if = { limit = { occupation_law = colonial_police_final } set_temp_variable = { occupation_casualty_factor = 1.0 } }           # Quasi-postcolonial administration
		if = { limit = { occupation_law = secretaries_general } set_temp_variable = { occupation_casualty_factor = 1.0 } }             # Semi-autonomous governorates

		### EXPLOIT BUT BUILD / SPECIAL CASES
		if = { limit = { occupation_law = princely_subjugation } set_temp_variable = { occupation_casualty_factor = 0.7 } }            # Puppet monarchies—moderate coercion, limited bloodshed

		### NO GARRISON (chaos, but decentralized violence)
		if = { limit = { occupation_law = no_garrison } set_temp_variable = { occupation_casualty_factor = 0.3 } }    
		
		multiply_temp_variable = { occupation_death_base = occupation_casualty_factor }

		# STEP 6: Apply warfare conduct civilian multiplier
		# Set by bm_conduct_* decisions. Default 1.0 (Standard).
		# Pacifist=0.5, Restrained=0.75, Standard=1.0, Aggressive=1.25, TotalWar=1.5
		if = {
			limit = { NOT = { check_variable = { controller.warfare_stance_civ_mult = 0 } } }
			multiply_temp_variable = { occupation_death_base = controller.warfare_stance_civ_mult }
		}
		
		# STEP 7: Generate casualties with controlled randomness (±30% variance)
		# Min = base * 0.7
		# Max = base * 1.3
		set_temp_variable = { temp_min = occupation_death_base }
		multiply_temp_variable = { temp_min = 0.7 }
		
		set_temp_variable = { temp_max = occupation_death_base }
		multiply_temp_variable = { temp_max = 1.3 }
		
		# Ensure minimum of at least 1 death if there's any resistance
		if = {
			limit = { check_variable = { temp_min < 1 } }
			set_temp_variable = { temp_min = 1 }
		}
		if = {
			limit = { check_variable = { temp_max < 2 } }
			set_temp_variable = { temp_max = 2 }
		}
		
		generate_random_casualties = yes
		
		# Optional: Log for debugging (uncomment to test)
		# log = "[THIS.GetName]: Occupation deaths - Pop:[?state_population_k]k, Resist:[?resistance]%, Dev:[?state_devastation_value]%, Deaths:[?temp_min]-[?temp_max]/week"
	}

	calc_pop_change = yes
	
}

	remove_from_array = { global.countries_casualty = 0 }	# dummy

casualty_sort_list_countries = {
	clear_array = global.countries_casualty
	
	every_country = {
		if = {
			limit = {
				NOT = { is_in_array = { global.countries_casualty = THIS.id } }
			}
			add_to_array = { global.countries_casualty = THIS.id }
		}
	}	

	casualty_sort_add_dummy = yes
}

casualty_sort_total = {
	casualty_sort_list_countries = yes
	casualty_sort_remove_dummy = yes
	
	clr_country_flag = casualty_sort_civ_descending
	clr_country_flag = casualty_sort_mil_descending
	clr_country_flag = casualty_sort_total_ascending
	clr_country_flag = casualty_sort_civ_ascending
	clr_country_flag = casualty_sort_mil_ascending
	
	if = {
		limit = {
			has_country_flag = casualty_sort_total_descending
		}
		clr_country_flag = casualty_sort_total_descending
		set_country_flag = casualty_sort_total_ascending
	}
	else = {
		set_country_flag = casualty_sort_total_descending
	}
	
	set_temp_variable = { len = global.countries_casualty^num }
	subtract_from_temp_variable = { len = 1 }
	##log = "Sort: Indexing Length is [?len]"
	set_temp_variable = { sorted = 0 }
	##log = "Sort: Is sorted - [?sorted]"
	
	while_loop_effect = {
		limit = { 
			NOT = { check_variable = { sorted = 1 } }
		}
		
		# Assume the array is sorted
		set_temp_variable = { sorted = 1 }
		##log = "[?sorted] While started"
		
		if = {
			limit = {
				has_country_flag = casualty_sort_total_descending
			}
			for_loop_effect = {
				start = 0
				end = len
				value = v
				
				set_temp_variable = { w = v }
				add_to_temp_variable = { w = 1 }
				
				if = {
					limit = {
						var:global.countries_casualty^v = {
							set_temp_variable = { global.temp_a = total_casualties_k }
						}
						var:global.countries_casualty^w = {
							set_temp_variable = { global.temp_b = total_casualties_k }
						}
						check_variable = { global.temp_a < global.temp_b }
					}
					set_temp_variable = { sorted = 0 }
					
					##log = "[?global.countries_casualty^v.GetTag] swapped with [?global.countries_casualty^w.GetTag]"
					
					set_temp_variable = { global.sort_a = global.countries_casualty^v }
					set_variable = { global.countries_casualty^v = global.countries_casualty^w }
					set_variable = { global.countries_casualty^w = global.sort_a }
				}
			}
		}
		else = {
			for_loop_effect = {
				start = 0
				end = len
				value = v
				
				set_temp_variable = { w = v }
				add_to_temp_variable = { w = 1 }
				
				if = {
					limit = {
						var:global.countries_casualty^v = {
							set_temp_variable = { global.temp_a = total_casualties_k }
						}
						var:global.countries_casualty^w = {
							set_temp_variable = { global.temp_b = total_casualties_k }
						}
						check_variable = { global.temp_a > global.temp_b }
					}
					set_temp_variable = { sorted = 0 }
					
					##log = "[?global.countries_casualty^v.GetTag] swapped with [?global.countries_casualty^w.GetTag]"
					
					set_temp_variable = { global.sort_a = global.countries_casualty^v }
					set_variable = { global.countries_casualty^v = global.countries_casualty^w }
					set_variable = { global.countries_casualty^w = global.sort_a }
				}
			}
		}
	}
	
	casualty_sort_add_dummy = yes
}
