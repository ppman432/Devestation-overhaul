		OR = {

			state_is_combat_bubble = yes
		}
	}
}

state_is_partially_occupied = {
	custom_trigger_tooltip = {
		tooltip = state_is_partially_occupied_tt
		
		NOT = {
			OR = {
				is_fully_controlled_by = controller
				is_fully_controlled_by = owner
			}
		}
		
		check_variable = { var = damaged_building_level@infrastructure value = 0 compare = greater_than }
	}
}

state_is_besieged = {
    # Must be at war
    controller = { has_war = yes }
    
    # Must have land neighbors (not an island)
    any_neighbor_state = { always = yes }
    
    # ALL land neighbors must be ENEMY-controlled
    # (i.e., no friendly OR neutral neighbors)
    NOT = {
        any_neighbor_state = {
            OR = {
                # Neighbor controlled by same country
                controller = { tag = PREV.PREV.controller }
                
                # Neighbor controlled by ally
                controller = { is_in_faction_with = PREV.PREV.controller }
                
                # Neighbor controlled by a neutral (not at war with us)
                controller = {
                    NOT = { has_war_with = PREV.PREV.controller }
                }
            }
        }
    }
}
